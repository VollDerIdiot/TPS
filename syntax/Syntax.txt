TPS: Tolle Programmier Sprache

Datei:
	[Dateikopf]
	[Sache]+
Dateikopf:
	[ort]
	[braucht]
Sache:
	[Sachenkopf]
	[Sacheninhalt]

Sachenart:
	[DingArt]
	[DingPlanArt]
	[UnfertigesDingArt]
	[KlassenArt]
	[KlassenPlanArt]
	[UnfertigeKlasseArt]

Sichtbarkeit:
	[offen]
	[vererben]
	[datei]
	[eigen]

Name:
	[RohName] (Potentiell:		[Zusatzgedings])

RohName:
	[Buchstabe][Buchstabe]+ 		(Immer mindestens 2 Buchstaben)

Buchstabe:
	[Großbuchstabe / Kleinbuchstabe]

Kleinbuchstabe:
	a-z, ö, ä, ü, ß

Großbuchstabe:
	A-Z, Ö, Ä, Ü, ẞ

Zahl:
	- Negativezahl:
		NEG-[PosZahl]
	- Positivezahl:
		[PosZahl]

PosZahl:
	- Decimal:
		DEC:(0-9)+
	- Hexadecimal:
		HEX:(0-9|A-F)+
	- Binär:
		BIN:(0-1)+

Pfad:
	startet mit '|:' 
	pfad mit '/' als Trennzeichen für Ordner
	endet mit ':>[ZEILENUMBRUCH]' (es sollte das letzte vor Zeilenende genommen werden)

Befehl:
	- [Methodenaufruf]
	- [Schleife]
	- [Falls]
	- [SetzeAuf]
